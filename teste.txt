import os

clear = lambda: os.system('cls')


def Menu(titulo, opcoes, nop):
    clear()
    ver()
    ver(titulo)
    ver()
    para i in alcance(nop):
        ver(i + 1, "-", opcoes[i])
        ver("-" * 30)
    ver("0 - Terminar")
    enquanto verdadeiro:
        tenta:
            op = int(ler("Opcao? "))
        excepto ValueError:
            ver("Digite um numero inteiro!")
            continuar
        se op >= 0 and op <= nop:
            parar
        entao:
            ver("O valor deve estar entre 1 e 4")

    devolve op
